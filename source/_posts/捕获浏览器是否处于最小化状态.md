```
function isMinStatus() { 
	var isMin = false; 
	if (window.outerWidth != undefined) { 
		isMin = window.outerWidth <= 160 && window.outerHeight <= 27; 
	} 
	else { 
		isMin = window.screenTop < -30000 && window.screenLeft < -30000; 
	} 
	return isMin; 
}
```
